#include <stdio.h>
#include <stdlib.h>

int which_big(const void *a,const void *b) {
    int num1=*(int*)a;
    int num2=*(int*)b;
    if (num1>num2) {    //1일경우 두수를 바꿈
        return 1;
    }
    else if (num1<num2) {
        return -1;
    }
    else {
        return 0;
    }
}

int main(void) {
    int n;
    int input[1001];
    scanf("%d",&n);
    
    for (int i=0; i<n; i++) {
        scanf("%d",&input[i]);
    }
    
    qsort(input,n,sizeof(int),which_big);
    
    for (int i=0; i<n; i++) {
        printf("%d\n",input[i]);
    }
    
    return 0;
}

비교함수 만들기

const void * 쓰기(float,char,구조체까지 정렬할수있어서 어떤 자료형이든
처리하기 위해 void을 씀)
(int*)로 강제 형변환을함 ( void*타입이어서 바로 꺼내어서 쓸수없음)

return1이면 둘의 순서를 바꿈 (양수이면됨)
return-1이면 순서를 그대로둠 (음수이면됨)
return0이면 순서를 그대로둠(두 값이 같음)

qsort은 4개의 인자를 받음
qsort(배열의 시작주소,배열에 들어있는 요소의 개수,배열 요소 하나하나의
크기  sizeof(int)같이,비교함수의 이름)
